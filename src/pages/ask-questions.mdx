let chatActive = false;

window.toggleChat = function() {
  const button = document.getElementById('chatToggle');
  const avatar = document.getElementById('heygen-streaming-embed');
  
  if (!chatActive) {
    // Start chat
    button.textContent = 'End Conversation';
    button.className = 'bg-red-600 hover:bg-red-700 text-white px-6 py-3 rounded-lg transition-colors font-medium';
    chatActive = true;
    
    // Simulate clicking the avatar to start
    if (avatar) {
      avatar.click();
    }
  } else {
    // End chat
    location.reload();
  }
}
`}
